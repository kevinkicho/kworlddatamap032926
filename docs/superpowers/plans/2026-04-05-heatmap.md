# Heatmap Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add per-metric Leaflet.heat gradient overlays to the map, wired into the filter panel with a heatmap toggle button per metric and a Show/Dim/Hide dot-mode row.

**Architecture:** Three new globals (`_heatmapMetric`, `_heatmapLayer`, `_heatDotMode`) control heatmap state. `setHeatmapMetric(metric)` builds normalised point arrays using the 95th-percentile cap and creates/updates the Leaflet.heat layer. `rebuildMapLayer()` checks heatmap state to override dot colour/opacity. Heatmap toggle buttons live inside the filter panel value-filter groups; dot-mode row appears below when a heatmap is active.

**Tech Stack:** Leaflet.heat 0.2.0 (CDN), Vanilla JS, CSS transitions matching existing filter panel styles.

```js
function _buildHeatPoints(metric) {
  // Gather raw values
  const raw = [];
  for (const city of allCities) {
    const qid = city.qid;
    let val = null;
    if (metric === 'nobel')        val = nobelCitiesData[qid]?.total ?? 0;
    else if (metric === 'universities') val = universitiesData[qid]?.length ?? 0;
    else if (metric === 'pop')     val = city.pop || 0;
    else if (metric === 'metro')   val = metroTransitData[qid]?.stations ?? 0;
    else if (metric === 'aq') {
      // AQ: pm25 value, higher = worse = hotter (not inverted in value, gradient handles it)
      val = airQualityData[qid]?.pm25 ?? null;
    }
    if (val !== null && val > 0) raw.push({ lat: city.lat, lng: city.lng, val });
  }
  if (raw.length === 0) return [];
  // 95th-percentile cap to prevent outlier washout
  const sorted = raw.map(r => r.val).sort((a, b) => a - b);
  const p95 = sorted[Math.floor(sorted.length * 0.95)] || sorted[sorted.length - 1];
  return raw.map(r => [r.lat, r.lng, Math.min(r.val / p95, 1.0)]);
}

function _heatGradient(metric) {
  const gradients = {
    nobel:        { 0.2: '#1a1a4e', 0.5: '#7b2ff7', 1: '#bc8cff' },
    universities: { 0.2: '#0d2b4e', 0.5: '#1f6feb', 1: '#a5d6ff' },
    pop:          { 0.2: '#0d3b2e', 0.5: '#20c997', 1: '#e6fffa' },
    metro:        { 0.2: '#2d1a00', 0.5: '#f0a500', 1: '#fff3cd' },
    aq:           { 0.0: '#3fb950', 0.4: '#f0a500', 0.7: '#f85149', 1: '#bc8cff' },
  };
  return gradients[metric] || gradients.pop;
}
```

```js
async function setHeatmapMetric(metric) {
  // If same metric clicked again, clear the heatmap
  if (_heatmapMetric === metric) {
    clearHeatmap();
    return;
  }
  // Ensure data is loaded for metrics that are lazy
  if (metric === 'aq')           await _ensureAirQuality();
  if (metric === 'metro')        await _ensureMetroTransit();
  if (metric === 'nobel')        await _ensureNobelCities();
  if (metric === 'universities') await _ensureUniversities();

  _heatmapMetric = metric;

  // Update toggle button active states
  document.querySelectorAll('.filter-heat-btn').forEach(b => {
    b.classList.toggle('active', b.dataset.metric === metric);
  });

  // Show dot-mode row
  const dotRow = document.getElementById('heat-dot-row');
  if (dotRow) dotRow.style.display = 'flex';

  // Build points and create/replace heat layer
  const points = _buildHeatPoints(metric);
  if (_heatmapLayer) { _heatmapLayer.remove(); _heatmapLayer = null; }
  if (points.length > 0 && typeof L !== 'undefined' && L.heatLayer) {
    _heatmapLayer = L.heatLayer(points, {
      radius: 35, blur: 25, maxZoom: 10, max: 1.0,
      gradient: _heatGradient(metric),
    }).addTo(map);
  }

  // Rebuild dots to apply dim/hide mode
  rebuildMapLayer();
}

function clearHeatmap() {
  _heatmapMetric = null;
  if (_heatmapLayer) { _heatmapLayer.remove(); _heatmapLayer = null; }
  // Reset dot-mode row
  const dotRow = document.getElementById('heat-dot-row');
  if (dotRow) dotRow.style.display = 'none';
  // Reset all heatmap toggle buttons
  document.querySelectorAll('.filter-heat-btn').forEach(b => b.classList.remove('active'));
  // Restore dots
  rebuildMapLayer();
}

function setHeatDotMode(mode) {
  _heatDotMode = mode;
  document.querySelectorAll('[data-dotmode]').forEach(b => {
    b.classList.toggle('active', b.dataset.dotmode === mode);
  });
  rebuildMapLayer();
}
```
